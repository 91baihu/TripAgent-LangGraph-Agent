**地图渲染函数（使用项目主色 `#12B7F5`）：**

```python
import folium
from streamlit_folium import st_folium

def render_route_map(trace):
    """从工具调用 trace 中提取路线数据，用 folium 渲染"""
    spots = extract_route_spots(trace)
    if not spots:
        st.info("发送旅行需求后，路线地图将在此显示")
        return

    center = [spots[0]["lat"], spots[0]["lng"]]
    m = folium.Map(location=center, zoom_start=13, tiles="OpenStreetMap")

    # 用项目配色绘制 Marker 和连线
    colors = ["#12B7F5", "#0E9FD6", "#07C160"]  # primary → primary-hover → success
    prev = None
    for i, spot in enumerate(spots):
        color = colors[i % len(colors)]
        # Marker：编号圆点
        folium.CircleMarker(
            location=[spot["lat"], spot["lng"]],
            radius=8,
            popup=f"<b>{i+1}. {spot['name']}</b>",
            color=color,
            fill=True,
            fill_color=color,
            fill_opacity=0.8,
        ).add_to(m)
        # 数字标签
        folium.Marker(
            location=[spot["lat"], spot["lng"]],
            icon=folium.DivIcon(
                html=f'<div style="color:white;font-weight:bold;font-size:12px">{i+1}</div>'
            ),
        ).add_to(m)

        if prev:
            folium.PolyLine(
                locations=[[prev["lat"], prev["lng"]], [spot["lat"], spot["lng"]]],
                color=color,
                weight=3,
                dash_array="5,5",
                opacity=0.7,
            ).add_to(m)
        prev = spot

    st_folium(m, height=350, use_container_width=True)
```

```python
import pandas as pd

def render_ranking_table(trace):
    """从工具调用 trace 中提取餐厅数据，渲染排名表格"""
    items = extract_restaurant_items(trace)
    if not items:
        st.info("发送旅行需求后，餐厅排行榜将在此显示")
        return

    # 用 pandas DataFrame + 条件着色
    df = pd.DataFrame(items)
    df_display = df[["rank", "name", "rating", "price_per_person", "distance_m", "type"]]
    df_display.columns = ["排名", "餐厅", "评分", "人均(¥)", "距离(m)", "类型"]

    # 条件着色：🥇金色、🥈银色、🥉铜色
    def highlight_rank(val):
        if val == 1:
            return "background-color: #FFF7E6; font-weight: bold"  # 金色底
        elif val == 2:
            return "background-color: #F0F1F3; font-weight: bold"  # 银灰底
        elif val == 3:
            return "background-color: #FFF0E0; font-weight: bold"  # 铜色底
        return ""

    styled = df_display.style.applymap(highlight_rank, subset=["排名"])
    st.dataframe(styled, height=350, use_container_width=True, hide_index=True)
```

```typescript
// 高德地图 Hook
// 配色映射：景点 Marker = primary #12B7F5, 路线 = primary-hover #0E9FD6, 餐厅 = success #07C160
import { useEffect, useRef, useState, useCallback } from "react";

const COLORS = {
  spot: "#12B7F5",           // 景点标记 → primary
  route: "#0E9FD6",          // 路线 → primary-hover
  walk: "#07C160",           // 步行路线 → semantic-success
  restaurant: "#FA9D3B",      // 餐厅标记 → semantic-warning
  hotel: "#576B95",           // 酒店标记 → semantic-link
};

interface Spot {
  name: string;
  lat: number;
  lng: number;
  order: number;
}

interface Route {
  from: Spot;
  to: Spot;
  distance_km: number;
  transport: string;
}

export function useAmap(containerId: string) {
  const mapRef = useRef<any>(null);
  const [loaded, setLoaded] = useState(false);

  // 初始化地图
  useEffect(() => {
    const init = () => {
      if (!(window as any).AMap) return;
      const map = new (window as any).AMap.Map(containerId, {
        zoom: 12,
        center: [116.3972, 39.9163], // 默认北京
        mapStyle: "amap://styles/light",
      });
      mapRef.current = map;
      setLoaded(true);
    };
    init();
  }, [containerId]);

  // 添加景点标记
  const addSpotMarker = useCallback((spot: Spot) => {
    if (!mapRef.current) return;
    const marker = new (window as any).AMap.Marker({
      position: [spot.lng, spot.lat],
      content: `
        <div style="
          background: ${COLORS.spot}; color: white;
          width: 28px; height: 28px; border-radius: 14px;
          display: flex; align-items: center; justify-content: center;
          font-size: 12px; font-weight: 700;
          box-shadow: 0 2px 6px rgba(18,183,245,0.3);
          border: 2px solid white;
        ">${spot.order}</div>
      `,
      offset: new (window as any).AMap.Pixel(-14, -14),
    });
    marker.setMap(mapRef.current);
    // 点击弹窗
    marker.on("click", () => {
      const info = new (window as any).AMap.InfoWindow({
        content: `<div style="padding:4px 8px;font-size:14px;font-family:'PingFang SC'"><b>${spot.name}</b></div>`,
        offset: new (window as any).AMap.Pixel(0, -35),
      });
      info.open(mapRef.current, [spot.lng, spot.lat]);
    });
    return marker;
  }, []);

  // 绘制路线（步行=绿实线, 驾车=蓝虚线, 地铁=蓝实线）
  const addRoute = useCallback((from: Spot, to: Spot, transport: string) => {
    if (!mapRef.current) return;
    const colorMap: Record<string, string> = {
      "步行": COLORS.walk,
      "驾车": COLORS.route,
      "打车": COLORS.route,
      "地铁": COLORS.route,
      "公交": COLORS.route,
    };
    const color = colorMap[transport] || COLORS.route;
    const dash = transport === "步行" ? "solid" : "dashed";

    const polyline = new (window as any).AMap.Polyline({
      path: [[from.lng, from.lat], [to.lng, to.lat]],
      strokeColor: color,
      strokeWeight: 3,
      strokeOpacity: 0.7,
      strokeDasharray: dash === "dashed" ? [10, 5] : undefined,
      showDir: true,
    });
    polyline.setMap(mapRef.current);
    return polyline;
  }, []);

  return { map: mapRef, loaded, addSpotMarker, addRoute };
}
```

## 🎨 设计一致性检查清单

每个新组件必须遵循：

- [x] **颜色**：仅使用 design-tokens 定义的色值，不引入新颜色
- [x] **间距**：遵循 8px 体系（m-1=4px, m-2=8px, m-3=12px, m-4=16px...）
- [x] **圆角**：卡片用 `rounded-card`(12px)，按钮用 `rounded-button`(10px)
- [x] **字体**：使用 Tailwind `text-h1/h2/h3/body/caption/small`
- [x] **阴影**：卡片 `shadow-card`，按钮 `shadow-button`
- [x] **动效**：`card-enter`（0.25s ease-out 从下淡入），`bubble-enter`（0.2s ease-out 缩放）
- [x] **组件复用**：优先使用 Card、Tag、Button，不重复造轮子
- [x] **无障碍**：`prefers-reduced-motion` 时关闭动效
