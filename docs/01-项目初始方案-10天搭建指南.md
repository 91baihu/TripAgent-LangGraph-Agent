```python
from langgraph.graph.message import add_messages
from langchain_core.messages import BaseMessage

class AgentState(TypedDict):
```

#### Step 5：构建 Agent Graph（核心）

```python
# src/agent/graph.py
from langgraph.graph import StateGraph, END
from langgraph.prebuilt import ToolNode
from langchain_deepseek import ChatDeepSeek
from langchain_core.messages import SystemMessage
import os

from .state import AgentState
from .prompts import SYSTEM_PROMPT
from ..tools.attractions import search_attractions


def create_agent():
    # 1. 绑定工具到 LLM
    tools = [search_attractions]
    llm = ChatDeepSeek(
        model="deepseek-chat",
        api_key=os.getenv("DEEPSEEK_API_KEY"),
        temperature=0.3
    )
    llm_with_tools = llm.bind_tools(tools)

    # 2. 定义节点
    def agent_node(state: AgentState):
        """Agent 思考节点：决定下一步做什么"""
        messages = state["messages"]
        # 首次调用时注入 system prompt
        if not any(isinstance(m, SystemMessage) for m in messages):
            messages = [SystemMessage(content=SYSTEM_PROMPT)] + messages
        response = llm_with_tools.invoke(messages)
        return {"messages": [response]}

    tool_node = ToolNode(tools)  # LangGraph 内置工具执行节点

    # 3. 路由函数：判断继续还是结束
    def should_continue(state: AgentState):
        last_message = state["messages"][-1]
        # 如果 LLM 想调工具 → 去 tool_node
        if hasattr(last_message, "tool_calls") and last_message.tool_calls:
            return "tools"
        # 否则结束
        return END

    # 4. 构建图
    workflow = StateGraph(AgentState)
    workflow.add_node("agent", agent_node)
    workflow.add_node("tools", tool_node)

    workflow.set_entry_point("agent")
    workflow.add_conditional_edges("agent", should_continue, {
        "tools": "tools",
        END: END
    })
    workflow.add_edge("tools", "agent")  # 工具结果 → 回到 agent 继续思考

    return workflow.compile()
```

#### Step 6：Streamlit 前端

```python
# src/main.py
import streamlit as st

st.set_page_config(page_title="TripAgent", page_icon="✈️")

    with st.chat_message(msg["role"]):
        st.markdown(msg["content"])

# 用户输入

    st.session_state.messages.append({"role": "user", "content": prompt})
    with st.chat_message("user"):
        st.markdown(prompt)

    with st.chat_message("assistant"):
        with st.spinner("Agent 思考中..."):
            result = st.session_state.agent.invoke({
                "messages": [{"role": "user", "content": prompt}],
                "next_step": "",
                "travel_plan": {}
            })
            reply = result["messages"][-1].content
            st.markdown(reply)

    st.session_state.messages.append({"role": "assistant", "content": reply})
```

```python
# src/tools/weather.py
import requests
from langchain_core.tools import tool

@tool
def get_weather(city: str, date: str = "today") -> str:
    """查询指定城市和日期的天气。规划行程时先查天气，避免雨天安排户外景点。

    Args:
        city: 城市名英文，如 'Beijing'、'Hangzhou'
        date: 日期，如 '2026-07-10' 或 'today'
    """
    try:
        url = f"https://wttr.in/{city}?format=%C+%t+%h+%w&lang=zh"
        resp = requests.get(url, timeout=5)
        return f"{city} 天气：{resp.text}"
    except:
        return f"无法获取{city}天气，建议出行前查看天气预报"
```

#### 路线规划工具

```python
# src/tools/route.py
from langchain_core.tools import tool
import math

# 模拟城市坐标（后续可接高德/百度地图API）
CITY_COORDS = {
    "故宫": (39.9163, 116.3972),
    "颐和园": (39.9999, 116.2755),
    "中国科技馆": (39.9747, 116.3933),
    "北京动物园": (39.9423, 116.3364),
    "798艺术区": (39.9842, 116.4951),
    "西湖": (30.2405, 120.1432),
    "灵隐寺": (30.2427, 120.1013),
    "杭州乐园": (30.2284, 120.2485),
}


@tool
def plan_route(spot_a: str, spot_b: str) -> str:
    """计算两个景点之间的距离和推荐交通方式。安排每日行程顺序时使用。

    Args:
        spot_a: 起点景点名称
        spot_b: 终点景点名称
    """
    coord_a = CITY_COORDS.get(spot_a)
    coord_b = CITY_COORDS.get(spot_b)

    if not coord_a or not coord_b:
        return f"未找到 '{spot_a}' 或 '{spot_b}' 的坐标信息"

    # 简化的距离计算
    lat_diff = (coord_b[0] - coord_a[0]) * 111
    lng_diff = (coord_b[1] - coord_a[1]) * 111 * math.cos(math.radians(coord_a[0]))
    distance_km = math.sqrt(lat_diff**2 + lng_diff**2)

    if distance_km < 3:
        transport = "步行约15分钟"
    elif distance_km < 10:
        transport = f"地铁约{distance_km * 2:.0f}分钟"
    else:
        transport = f"打车约{distance_km * 1.5:.0f}分钟"

    return f"{spot_a} → {spot_b}：直线距离约{distance_km:.1f}km，建议{transport}"
```

#### 接入景小游 RAG（升级版景点工具）

```python
# src/tools/attractions.py — 升级版
# 把景小游的 rag/retriever.py 复制到 trip-agent/src/rag/
from ..rag.retriever import HybridRetriever

# 初始化一次
retriever = HybridRetriever(
    bm25_weight=0.3,
    vector_weight=0.7,
    embedding_model="BAAI/bge-small-zh-v1.5"
)

@tool
def search_attractions(city: str, keyword: str = "") -> str:
    """搜索指定城市的景点信息。当用户询问某地有什么好玩的、某个景点的详情时调用。

    Args:
        city: 城市名，如 '北京'、'杭州'
        keyword: 搜索关键词，如 '亲子'、'博物馆'、'自然风光'
    """
    query = f"{city} {keyword} 景点介绍"
    results = retriever.search(query, top_k=5)

    if not results:
        return f"未找到{city}相关的景点信息，请尝试其他关键词"

    formatted = f"## {city} 景点搜索结果\n\n"
    for i, doc in enumerate(results, 1):
        formatted += f"{i}. **{doc.get('title', '景点')}**\n"
        formatted += f"   {doc.get('content', '')[:200]}...\n\n"
    return formatted
```

```python
                # 使用 stream 模式捕获中间步骤
                trace = []
                step_num = 0
                for event in st.session_state.agent.stream(
                    {"messages": [{"role": "user", "content": prompt}],
                     "next_step": "", "travel_plan": {}},
                    stream_mode="values"
                ):
                    last_msg = event["messages"][-1]
                    if isinstance(last_msg, AIMessage) and hasattr(last_msg, "tool_calls") and last_msg.tool_calls:
                        for tc in last_msg.tool_calls:
                            step_num += 1
                            trace.append({
                                "num": step_num,
                                "action": "调用工具",
                                "tool": tc["name"],
                                "input": tc["args"]
                            })
                    elif isinstance(last_msg, ToolMessage):
                        trace[-1]["output"] = last_msg.content[:300]
                    elif isinstance(last_msg, AIMessage) and last_msg.content:
                        step_num += 1
                        trace.append({
                            "num": step_num,
                            "action": "思考/回复",
                            "thought": last_msg.content[:200]
                        })

                final_reply = event["messages"][-1].content
                st.session_state.trace = trace
                st.markdown(final_reply)

        st.session_state.messages.append({"role": "assistant", "content": final_reply})

# === 右侧：推理追踪面板 ===
with col_sidebar:
    st.subheader("🔍 Agent 推理过程")
    if st.session_state.trace:
        for step in st.session_state.trace:
            with st.expander(f"Step {step['num']}: {step['action']}"):
                if step["action"] == "调用工具":
                    st.caption(f"🔧 工具: {step['tool']}")
                    st.caption("📥 参数:")
                    st.json(step["input"])
                    if "output" in step:
                        st.caption("📤 返回:")
                        st.text(step["output"])
                elif step["action"] == "思考/回复":
                    st.text(step.get("thought", ""))
    else:
        st.caption("发送一条消息来观察 Agent 的推理过程...")
```

**Day 5-6 完成标志：** 发送消息后，右侧面板实时展示每一步推理——调了什么工具、传了什么参数、返回了什么结果。

```python
# src/tools/weather.py
import requests
from langchain_core.tools import tool

@tool
def get_weather(city: str, date: str = "today") -> str:
    """查询指定城市和日期的天气..."""
    try:
        url = f"https://wttr.in/{city}?format=%C+%t+%h+%w&lang=zh"
        resp = requests.get(url, timeout=5)
        resp.raise_for_status()
        return f"{city} 天气：{resp.text}"
    except requests.Timeout:
        return f"天气服务响应超时，建议稍后重试。{city} 当前季节一般为晴好天气。"
    except requests.RequestException:
        return f"天气服务暂不可用。建议出行前在手机天气 App 查看{city}最新天气。"
```

## 🎙️ 面试话术

### Agent 问题

> **面试官：** "你了解 Agent 吗？"

> **你：** "了解，我在一个旅行规划项目里实践过。传统的 RAG 对话是被动的——用户问一句它答一句。Agent 的区别是它能主动多步推理。比如用户说「帮我规划 3 天北京亲子游」，我的 Agent 会先理解意图，然后自己决定：先调景点搜索工具拿到适合亲子的景点列表→再查那几天的天气→再算景点之间的距离排顺序→最后综合输出一个 Day1/Day2/Day3 的完整行程。这个过程不是硬编码的，是 LangGraph 的 StateGraph 驱动的——每一步 Agent 都会判断'我的信息够不够输出最终答案，还是需要再调一个工具'。你要不要看下我的推理过程可视化界面？"

### LangChain/LangGraph 问题

> **面试官：** "为什么用 LangGraph 而不是直接调 API？"

> **你：** "直接调 API 只能做单轮问答。LangGraph 解决的是'多步推理'的问题——它用状态图把 Agent 的思考过程建模成节点和边：agent 节点负责思考，tools 节点负责执行，中间有一条条件边判断'信息够了就结束，不够就继续调工具'。这个 ReAct 循环（Reasoning + Acting）是 LangGraph 帮我管理的，我不需要手写 while 循环。"

### RAG vs Agent 的区别

> **面试官：** "你两个项目都做了，RAG 和 Agent 什么区别？"

> **你：** "景小游是 RAG——用户问「故宫几点关门」，系统检索知识库然后回答。TripAgent 是 Agent——用户说「帮我规划3天北京游」，Agent 需要自己决定'我应该先搜景点，然后查天气，再算路线，最后综合输出'。RAG 是检索+生成，Agent 是推理+工具调用+生成。实际业务里，Agent 通常内置 RAG 作为其中一个工具——我 TripAgent 的 search_attractions 工具就是复用了景小游的 RAG 模块。"
